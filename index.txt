    <script>
        // 定義伊登 12 色相環的精確顏色資料 (順時針，每 30 度一個顏色)
        const colorList = [
            { name: "黃色 (Yellow)", hex: "#FFFF00" },
            { name: "黃橙色 (Yellow-Orange)", hex: "#FFB300" },
            { name: "橙色 (Orange)", hex: "#FF6F00" },
            { name: "紅橙色 (Red-Orange)", hex: "#FF3D00" },
            { name: "紅色 (Red)", hex: "#D50000" },
            { name: "紅紫色 (Red-Violet)", hex: "#C51162" },
            { name: "紫色 (Violet)", hex: "#AA00FF" },
            { name: "藍紫色 (Blue-Violet)", hex: "#6200EA" },
            { name: "藍色 (Blue)", hex: "#2962FF" },
            { name: "藍綠色 (Blue-Green)", hex: "#00B8D4" },
            { name: "綠色 (Green)", hex: "#00C853" },
            { name: "黃綠色 (Yellow-Green)", hex: "#AEEA00" }
        ];

        let currentRotation = 0; // 記錄目前色相環轉到的總角度

        function mixColors() {
            const wheel = document.getElementById('wheel');
            const resultLabel = document.getElementById('resultLabel');
            const colorResult = document.getElementById('colorResult');
            
            const idx1 = parseInt(document.getElementById('color1').value);
            const idx2 = parseInt(document.getElementById('color2').value);
            
            // 隱藏前一次的結果色塊，做重新淡入動效
            colorResult.classList.remove('show');
            resultLabel.innerText = "🧪 正在神奇旋轉調色中...";

            // 1. 計算旋轉：每次點擊都在原本基礎上多轉 3~4 圈 (1080~1440度)，隨機增加動態感
            const randomSpin = 1080 + Math.floor(Math.random() * 360);
            currentRotation += randomSpin;
            wheel.style.transform = `rotate(${currentRotation}deg)`;

            // 2. 混色演算法（計算色相環上的最短路徑中間值）
            let diff = Math.abs(idx1 - idx2);
            let mixedIdx;

            if (diff <= 6) {
                // 一般情況，直接取兩者算術平均數
                mixedIdx = Math.round((idx1 + idx2) / 2);
            } else {
                // 跨越圓圈邊界（例如黃色0 與 黃綠11）取最短路徑平均
                mixedIdx = Math.round((idx1 + idx2 + 12) / 2) % 12;
            }

            const finalColor = colorList[mixedIdx];

            // 3. 等待旋轉動畫結束 (2.5秒) 後顯示結果
            setTimeout(() => {
                resultLabel.innerText = `✨ 混色結果：${finalColor.name}`;
                colorResult.style.background = finalColor.hex;
                colorResult.classList.add('show');
            }, 2500);
        }
    </script>
